bool BaseLevel::init()
{
	if (!Layer::init())
	{
		return false;
	}
	setLevel();
	initVariables();
	createPhysicsFromLayer("ground", GROUND_TAG);
	createPhysicsFromLayer("wall", WALL_TAG);
	createPhysicsFromLayer("winning", WINNING_AREA);
	createEnemies();
	createRespawnButton();
	createBackButton();
	scheduleUpdate();
	return true;
}
